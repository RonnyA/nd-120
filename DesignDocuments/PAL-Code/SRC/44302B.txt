IF (/TEST) EMD = Q2 * QO * CACT         ; CPU CYCLE TO BUS SET
            + EMD * CACT 1              ;       "          HOLD

            + EMD * IORQ * CC2 * /TERM  ; ) CPU READ, FETCH AND

IF (/TEST) DSTB = CGNT + CACT * /BDRY50 * /BDRY25 * /IORQ
